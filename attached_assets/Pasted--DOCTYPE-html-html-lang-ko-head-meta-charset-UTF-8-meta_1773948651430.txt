    <script src="https://cdnjs.cloudflare.com/ajax/libs/xlsx/0.18.5/xlsx.full.min.js"></script>
    <style>
        body { font-family: sans-serif; padding: 20px; }
        #drop_zone { border: 2px dashed #ccc; padding: 50px; text-align: center; margin-bottom: 20px; cursor: pointer; }
        table { width: 100%; border-collapse: collapse; margin-top: 20px; }
        th, td { border: 1px solid #ddd; padding: 8px; text-align: left; }
        th { background-color: #f2f2f2; }
        .controls { margin-bottom: 20px; }
    </style>
</head>
<body>
    <h2>업체별 발주서 맞춤 검색 시스템</h2>

    <div class="controls">
        <label for="companySelect">업체 선택:</label>
        <select id="companySelect">
            <option value="A">A사 (품목명 기준 검색)</option>
            <option value="B">B사 (품목코드 기준 검색)</option>
        </select>
    </div>

    <div id="drop_zone">엑셀 파일을 이곳에 드래그 앤 드롭 하세요.</div>

    <div class="controls">
        <label for="searchInput">검색어:</label>
        <input type="text" id="searchInput" placeholder="검색어 입력">
        <button id="searchBtn">검색</button>
        <button id="downloadBtn" style="display:none;">엑셀 다운로드</button>
    </div>

    <table id="resultTable" style="display:none;">
        <thead id="tableHead"></thead>
        <tbody id="tableBody"></tbody>
    </table>

    <script>
        let globalExcelData = [];
        let currentHeaders = [];

        const companyConfigs = {
            "A": { searchColumn: "품목명" },
            "B": { searchColumn: "품목코드" }
        };

        const dropZone = document.getElementById('drop_zone');
        const searchInput = document.getElementById('searchInput');
        const searchBtn = document.getElementById('searchBtn');
        const downloadBtn = document.getElementById('downloadBtn');
        const resultTable = document.getElementById('resultTable');
        const tableHead = document.getElementById('tableHead');
        const tableBody = document.getElementById('tableBody');
        const companySelect = document.getElementById('companySelect');

        dropZone.addEventListener('dragover', (e) => {
            e.preventDefault();
            dropZone.style.borderColor = 'blue';
        });

        dropZone.addEventListener('dragleave', (e) => {
            e.preventDefault();
            dropZone.style.borderColor = '#ccc';
        });

        dropZone.addEventListener('drop', (e) => {
            e.preventDefault();
            dropZone.style.borderColor = '#ccc';
            const files = e.dataTransfer.files;
            if (files.length > 0) {
                readExcel(files[0]);
            }
        });

        function readExcel(file) {
            const reader = new FileReader();
            reader.onload = (e) => {
                const data = new Uint8Array(e.target.result);
                const workbook = XLSX.read(data, { type: 'array' });
                const firstSheetName = workbook.SheetNames[0];
                const worksheet = workbook.Sheets[firstSheetName];
                const json = XLSX.utils.sheet_to_json(worksheet, { header: 1 });
                
                if(json.length > 0) {
                    currentHeaders = json[0];
                    globalExcelData = XLSX.utils.sheet_to_json(worksheet);
                    alert("데이터 로드 완료");
                }
            };
            reader.readAsArrayBuffer(file);
        }

        searchBtn.addEventListener('click', () => {
            const query = searchInput.value.trim();
            const selectedCompany = companySelect.value;
            const targetColumn = companyConfigs[selectedCompany].searchColumn;

            if (!query) return;

            const filteredData = globalExcelData.filter(row => {
                const cellValue = row[targetColumn];
                return cellValue && String(cellValue).includes(query);
            });

            renderTable(filteredData);
        });

        function renderTable(data) {
            tableHead.innerHTML = '';
            tableBody.innerHTML = '';

            if (data.length === 0) {
                resultTable.style.display = 'none';
                downloadBtn.style.display = 'none';
                return;
            }

            const trHead = document.createElement('tr');
            currentHeaders.forEach(header => {
                const th = document.createElement('th');
                th.textContent = header;
                trHead.appendChild(th);
            });
            tableHead.appendChild(trHead);

            data.forEach(row => {
                const tr = document.createElement('tr');
                currentHeaders.forEach(header => {
                    const td = document.createElement('td');
                    td.textContent = row[header] || '';
                    tr.appendChild(td);
                });
                tableBody.appendChild(tr);
            });

            resultTable.style.display = 'table';
            downloadBtn.style.display = 'inline-block';
        }

        downloadBtn.addEventListener('click', () => {
            const ws = XLSX.utils.table_to_sheet(resultTable);
            const wb = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(wb, ws, "검색결과");
            XLSX.writeFile(wb, "검색결과_다운로드.xlsx");
        });
    </script>
</body>
